private void CreateConfigTab(TabPage parent)

// Métodos helper para crear controles

}

{

private TextBox CreateTextBox(string value, EventHandler textChanged)
{
    var txt = new TextBox 
    { 
        Text = value,
        Width = 250,
        BackColor = Color.FromArgb(50, 50, 55), 
        ForeColor = Color.White,
        Font = new Font("Segoe UI", 10),
        BorderStyle = BorderStyle.FixedSingle
    };
    txt.TextChanged += textChanged;
    return txt;
}

private CheckBox CreateCheckBox(string text, bool isChecked, EventHandler checkedChanged)
{
    var chk = new CheckBox 
    { 
        Text = text, 
        AutoSize = true,
        Checked = isChecked,
        ForeColor = Color.White,
        Font = new Font("Segoe UI", 9),
        Margin = new Padding(0, 5, 0, 5)
    };
    chk.CheckedChanged += checkedChanged;
    return chk;
}

private NumericUpDown CreateNumeric(decimal value, decimal min, decimal max, EventHandler valueChanged)
{
    var num = new NumericUpDown 
    { 
        Value = value,
        Minimum = min,
        Maximum = max,
        Width = 80,
        BackColor = Color.FromArgb(50, 50, 55), 
        ForeColor = Color.White,
        Font = new Font("Segoe UI", 10)
    };
    num.ValueChanged += valueChanged;
    return num;
}

private FlowLayoutPanel CreateConfigRow(string label, Control control)
{
    var row = new FlowLayoutPanel 
    { 
        AutoSize = true, 
        FlowDirection = FlowDirection.LeftToRight, 
        WrapContents = false,
        Margin = new Padding(0, 0, 0, 15)
    };
    var lbl = CreateLabel(label);
    lbl.Width = 150;
    row.Controls.Add(lbl);
    row.Controls.Add(control);
    return row;
}

private FlowLayoutPanel CreateNumericRow(string label, out NumericUpDown numeric, decimal value, decimal min, decimal max, EventHandler valueChanged)
{
    var row = new FlowLayoutPanel 
    { 
        AutoSize = true, 
        FlowDirection = FlowDirection.LeftToRight, 
        WrapContents = false,
        Margin = new Padding(0, 0, 0, 10)
    };
    var lbl = CreateLabel(label);
    lbl.Width = 200;
    row.Controls.Add(lbl);
    numeric = CreateNumeric(value, min, max, valueChanged);
    row.Controls.Add(numeric);
    return row;
}

private FlowLayoutPanel CreateNumericRow(string label, ref NumericUpDown numericField, decimal value, decimal min, decimal max, EventHandler valueChanged)
{
    var row = CreateNumericRow(label, out var numeric, value, min, max, valueChanged);
    numericField = numeric;
    return row;
}

// Event handlers (mantener la lógica existente)
private void OnAutoModeChanged(object sender, EventArgs e)
{
    autoMode = chkAutoMode.Checked;
    SaveConfig();
    if (autoMode)
    {
        InitializeAutoMode();
        Log("🤖 Modo Automático ACTIVADO");
    }
    else
    {
        StopAutoMode();
        Log("🤖 Modo Automático DESACTIVADO");
    }
}

private void OnTurboModeChanged(object sender, EventArgs e)
{
    if (chkTurboMode.Checked)
    {
        maxParallelDownloads = 8;
        maxSimultaneousDownloads = 8;
        maxParallelSearches = 12;
        searchTimeout = 20;
        numParallelDownloads.Value = 8;
        Log("⚡ Modo Turbo ACTIVADO");
    }
    else
    {
        maxParallelDownloads = 3;
        maxSimultaneousDownloads = 3;
        maxParallelSearches = 3;
        searchTimeout = 30;
        numParallelDownloads.Value = 3;
        Log("🐢 Modo Normal");
    }
    numParallelDownloads.Enabled = chkTurboMode.Checked;
    SaveConfig();
}

private void OnPrioritySizeChanged(object sender, EventArgs e)
{
    priorityBySize = chkPriorityBySize.Checked;
    SaveConfig();
    if (priorityBySize)
    {
        lock (downloadQueueLock)
        {
            var pending = downloadQueue.Where(t => t.Status == DownloadStatus.Pending).OrderBy(t => t.File.SizeBytes).ToList();
            var others = downloadQueue.Where(t => t.Status != DownloadStatus.Pending).ToList();
            downloadQueue.Clear();
            downloadQueue.AddRange(others);
            downloadQueue.AddRange(pending);
        }
        Log("📊 Cola reordenada: archivos pequeños primero");
    }
}

private void BrowseFolder()
{
    using (var dialog = new FolderBrowserDialog())
    {
        dialog.SelectedPath = downloadDir;
        if (dialog.ShowDialog() == DialogResult.OK)
        {
            txtDownloadDir.Text = dialog.SelectedPath;
        }
    }
}

private void OnEditBlacklistClick(object sender, EventArgs e)
{
    try
    {
        if (!System.IO.File.Exists(blacklistAuthorsFile))
        {
            SaveBlacklistedAuthors();
        }
        Process.Start("notepad.exe", blacklistAuthorsFile);
        MessageBox.Show("Edita la blacklist en el Notepad.\n\n• Un autor por línea\n• Las líneas con # son comentarios", "Editar Blacklist", MessageBoxButtons.OK, MessageBoxIcon.Information);
        Task.Delay(1000).ContinueWith(_ => SafeBeginInvoke(() => LoadBlacklistedAuthors()));
    }
    catch (Exception ex)
    {
        MessageBox.Show($"Error: {ex.Message}", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
    }
}
