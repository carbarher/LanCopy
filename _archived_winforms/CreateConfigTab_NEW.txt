        private void CreateConfigTab(TabPage parent)
        {
            parent.BackColor = Color.FromArgb(18, 18, 18);
            
            // Panel con scroll
            var scrollPanel = new Panel 
            { 
                Dock = DockStyle.Fill, 
                AutoScroll = true,
                BackColor = Color.FromArgb(18, 18, 18),
                Padding = new Padding(20)
            };
            
            // FlowLayoutPanel para organizar verticalmente
            var flowLayout = new FlowLayoutPanel
            {
                Dock = DockStyle.Top,
                AutoSize = true,
                FlowDirection = FlowDirection.TopDown,
                WrapContents = false,
                BackColor = Color.FromArgb(18, 18, 18),
                Padding = new Padding(0, 0, 20, 0)
            };
            
            // Título
            var lblTitle = new Label 
            { 
                Text = "⚙️ Configuración", 
                AutoSize = true,
                ForeColor = Color.White, 
                Font = new Font("Segoe UI", 16, FontStyle.Bold),
                Margin = new Padding(0, 0, 0, 30)
            };
            flowLayout.Controls.Add(lblTitle);
            
            // === SECCIÓN: CUENTA ===
            var accountPanel = CreateConfigSection("CUENTA", Color.FromArgb(0, 120, 215));
            
            // Usuario
            var userRow = CreateConfigRow("Usuario Soulseek:");
            txtUsername = new TextBox { Width = 300, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Text = username };
            txtUsername.TextChanged += (s, e) => { username = txtUsername.Text; SaveConfig(); };
            userRow.Controls.Add(txtUsername);
            accountPanel.Controls.Add(userRow);
            
            // Contraseña
            var passRow = CreateConfigRow("Contraseña:");
            txtPassword = new TextBox { Width = 300, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Text = password, UseSystemPasswordChar = true };
            txtPassword.TextChanged += (s, e) => { password = txtPassword.Text; SaveConfig(); };
            passRow.Controls.Add(txtPassword);
            accountPanel.Controls.Add(passRow);
            
            // Carpeta de descargas
            var dirRow = CreateConfigRow("Carpeta descargas:");
            txtDownloadDir = new TextBox { Width = 300, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Text = downloadDir };
            txtDownloadDir.TextChanged += (s, e) => { downloadDir = txtDownloadDir.Text; SaveConfig(); };
            dirRow.Controls.Add(txtDownloadDir);
            
            var btnBrowse = new Button { Text = "📁", Width = 40, Height = 30, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, FlatStyle = FlatStyle.Flat, Margin = new Padding(5, 0, 0, 0) };
            btnBrowse.Click += (s, e) => 
            {
                using (var dialog = new FolderBrowserDialog())
                {
                    dialog.SelectedPath = downloadDir;
                    if (dialog.ShowDialog() == DialogResult.OK)
                    {
                        txtDownloadDir.Text = dialog.SelectedPath;
                    }
                }
            };
            dirRow.Controls.Add(btnBrowse);
            
            var btnOpenFolder = new Button { Text = "🗂️ Abrir", Width = 80, Height = 30, BackColor = Color.FromArgb(0, 120, 215), ForeColor = Color.White, FlatStyle = FlatStyle.Flat, Margin = new Padding(5, 0, 0, 0) };
            btnOpenFolder.FlatAppearance.BorderSize = 0;
            btnOpenFolder.Click += (s, e) => 
            {
                try
                {
                    if (System.IO.Directory.Exists(downloadDir))
                    {
                        Process.Start("explorer.exe", downloadDir);
                    }
                    else
                    {
                        MessageBox.Show($"La carpeta no existe:\n{downloadDir}\n\nCrea la carpeta primero.", "Carpeta no encontrada", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                    }
                }
                catch (Exception ex)
                {
                    MessageBox.Show($"Error al abrir carpeta:\n{ex.Message}", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
            };
            dirRow.Controls.Add(btnOpenFolder);
            accountPanel.Controls.Add(dirRow);
            
            flowLayout.Controls.Add(accountPanel);
            
            // === SECCIÓN: GEMINI AI ===
            var geminiPanel = CreateConfigSection("GEMINI AI (CLASIFICACIÓN)", Color.FromArgb(100, 200, 255));
            
            var keyRow = CreateConfigRow("🤖 API Key:");
            var txtGeminiKey = new TextBox 
            { 
                Width = 400, 
                Text = geminiApiKey, 
                BackColor = Color.FromArgb(60, 60, 60), 
                ForeColor = Color.White,
                PlaceholderText = "AIzaSy..."
            };
            txtGeminiKey.TextChanged += (s, e) =>
            {
                geminiApiKey = txtGeminiKey.Text.Trim();
                SaveConfig();
                if (!string.IsNullOrEmpty(geminiApiKey))
                {
                    InitializeGeminiClassifier();
                }
            };
            keyRow.Controls.Add(txtGeminiKey);
            geminiPanel.Controls.Add(keyRow);
            
            var btnRow = new FlowLayoutPanel { AutoSize = true, FlowDirection = FlowDirection.LeftToRight, Margin = new Padding(0, 5, 0, 5) };
            var btnGetGeminiKey = new Button 
            { 
                Text = "🌐 Obtener API Key GRATIS", 
                Width = 200, 
                Height = 35, 
                BackColor = Color.FromArgb(0, 150, 136), 
                ForeColor = Color.White, 
                FlatStyle = FlatStyle.Flat 
            };
            btnGetGeminiKey.FlatAppearance.BorderSize = 0;
            btnGetGeminiKey.Click += (s, e) =>
            {
                try
                {
                    Process.Start(new ProcessStartInfo
                    {
                        FileName = "https://makersuite.google.com/app/apikey",
                        UseShellExecute = true
                    });
                }
                catch (Exception ex)
                {
                    MessageBox.Show($"Error abriendo navegador:\n{ex.Message}", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
            };
            btnRow.Controls.Add(btnGetGeminiKey);
            geminiPanel.Controls.Add(btnRow);
            
            var lblGeminiInfo = new Label 
            { 
                Text = "ℹ️ Gemini AI clasifica libros por género (ciencia ficción, fantasía, etc.)\n" +
                       "   100% GRATIS • 60 requests/min • 1500 requests/día",
                AutoSize = true, 
                ForeColor = Color.FromArgb(150, 150, 150),
                Font = new Font("Segoe UI", 8),
                Margin = new Padding(0, 5, 0, 0)
            };
            geminiPanel.Controls.Add(lblGeminiInfo);
            
            flowLayout.Controls.Add(geminiPanel);
            
            // === SECCIÓN: OPCIONES ===
            var optionsPanel = CreateConfigSection("OPCIONES", Color.FromArgb(0, 120, 215));
            
            chkAutoConnect = new CheckBox { Text = "Conectar automáticamente al iniciar", AutoSize = true, ForeColor = Color.White, Margin = new Padding(0, 5, 0, 5) };
            chkAutoConnect.Checked = autoConnect;
            chkAutoConnect.CheckedChanged += (s, e) => { autoConnect = chkAutoConnect.Checked; SaveConfig(); };
            optionsPanel.Controls.Add(chkAutoConnect);
            
            chkTurboMode = new CheckBox { Text = "⚡ Modo Turbo (descargas paralelas)", AutoSize = true, ForeColor = Color.White, Checked = maxParallelDownloads > 1, Margin = new Padding(0, 5, 0, 5) };
            chkTurboMode.CheckedChanged += (s, e) => 
            { 
                maxParallelDownloads = chkTurboMode.Checked ? 3 : 1;
                numParallelDownloads.Enabled = chkTurboMode.Checked;
                SaveConfig(); 
            };
            optionsPanel.Controls.Add(chkTurboMode);
            
            var parallelRow = CreateConfigRow("  Descargas simultáneas:");
            numParallelDownloads = new NumericUpDown { Width = 70, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Minimum = 1, Maximum = 10, Value = maxParallelDownloads, Enabled = chkTurboMode.Checked };
            numParallelDownloads.ValueChanged += (s, e) => { maxParallelDownloads = (int)numParallelDownloads.Value; SaveConfig(); };
            parallelRow.Controls.Add(numParallelDownloads);
            optionsPanel.Controls.Add(parallelRow);
            
            var retriesRow = CreateConfigRow("🔄 Reintentos automáticos:");
            numMaxRetries = new NumericUpDown { Width = 70, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Minimum = 0, Maximum = 10, Value = maxRetries };
            numMaxRetries.ValueChanged += (s, e) => { maxRetries = (int)numMaxRetries.Value; SaveConfig(); };
            retriesRow.Controls.Add(numMaxRetries);
            var lblRetriesInfo = new Label { Text = "(0 = sin reintentos)", AutoSize = true, ForeColor = Color.Gray, Font = new Font("Segoe UI", 8), Margin = new Padding(10, 0, 0, 0) };
            retriesRow.Controls.Add(lblRetriesInfo);
            optionsPanel.Controls.Add(retriesRow);
            
            flowLayout.Controls.Add(optionsPanel);
            
            // === SECCIÓN: RED Y BÚSQUEDA ===
            var networkPanel = CreateConfigSection("RED Y BÚSQUEDA", Color.FromArgb(0, 120, 215));
            
            var timeoutRow = CreateConfigRow("Timeout (seg, 0=sin límite):");
            var numTimeout = new NumericUpDown { Width = 80, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Minimum = 0, Maximum = 999999, Value = searchTimeout };
            numTimeout.ValueChanged += (s, e) => { searchTimeout = (int)numTimeout.Value; SaveConfig(); };
            timeoutRow.Controls.Add(numTimeout);
            networkPanel.Controls.Add(timeoutRow);
            
            var responseRow = CreateConfigRow("Respuestas (0=sin límite):");
            var numResponseLimit = new NumericUpDown { Width = 80, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Minimum = 0, Maximum = 999999, Value = responseLimit };
            numResponseLimit.ValueChanged += (s, e) => { responseLimit = (int)numResponseLimit.Value; SaveConfig(); };
            responseRow.Controls.Add(numResponseLimit);
            networkPanel.Controls.Add(responseRow);
            
            var fileRow = CreateConfigRow("Archivos (0=sin límite):");
            var numFileLimit = new NumericUpDown { Width = 80, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Minimum = 0, Maximum = 999999, Value = fileLimit };
            numFileLimit.ValueChanged += (s, e) => { fileLimit = (int)numFileLimit.Value; SaveConfig(); };
            fileRow.Controls.Add(numFileLimit);
            networkPanel.Controls.Add(fileRow);
            
            var searchesRow = CreateConfigRow("Búsquedas simultáneas:");
            var numParallelSearches = new NumericUpDown { Width = 80, BackColor = Color.FromArgb(60, 60, 60), ForeColor = Color.White, Minimum = 1, Maximum = 10, Value = maxParallelSearches };
            numParallelSearches.ValueChanged += (s, e) => { maxParallelSearches = (int)numParallelSearches.Value; SaveConfig(); };
            searchesRow.Controls.Add(numParallelSearches);
            var lblSearchesInfo = new Label { Text = "(búsqueda múltiple con comas)", AutoSize = true, ForeColor = Color.Gray, Font = new Font("Segoe UI", 8), Margin = new Padding(10, 0, 0, 0) };
            searchesRow.Controls.Add(lblSearchesInfo);
            networkPanel.Controls.Add(searchesRow);
            
            var chkDistributed = new CheckBox { Text = "Habilitar red distribuida", AutoSize = true, ForeColor = Color.White, Checked = enableDistributedNetwork, Margin = new Padding(0, 10, 0, 10) };
            chkDistributed.CheckedChanged += (s, e) => { enableDistributedNetwork = chkDistributed.Checked; SaveConfig(); };
            networkPanel.Controls.Add(chkDistributed);
            
            flowLayout.Controls.Add(networkPanel);
            
            // === BOTONES DE ACCIÓN ===
            var actionsPanel = CreateConfigSection("ACCIONES", Color.FromArgb(0, 120, 215));
            
            var buttonsRow = new FlowLayoutPanel { AutoSize = true, FlowDirection = FlowDirection.LeftToRight, Margin = new Padding(0, 5, 0, 5) };
            
            var btnTestConnection = new Button { Text = "🔌 Test Conexión", Width = 160, Height = 40, BackColor = Color.FromArgb(0, 120, 215), ForeColor = Color.White, FlatStyle = FlatStyle.Flat, Font = new Font("Segoe UI", 10, FontStyle.Bold), Margin = new Padding(0, 0, 10, 0) };
            btnTestConnection.Click += async (s, e) => await TestConnection();
            buttonsRow.Controls.Add(btnTestConnection);
            
            var btnEditBlacklist = new Button { Text = "🚫 Editar Blacklist", Width = 160, Height = 40, BackColor = Color.FromArgb(150, 60, 60), ForeColor = Color.White, FlatStyle = FlatStyle.Flat, Font = new Font("Segoe UI", 10, FontStyle.Bold), Margin = new Padding(0, 0, 10, 0) };
            btnEditBlacklist.Click += (s, e) =>
            {
                try
                {
                    if (!System.IO.File.Exists(blacklistAuthorsFile))
                    {
                        SaveBlacklistedAuthors();
                    }
                    
                    Process.Start("notepad.exe", blacklistAuthorsFile);
                    
                    MessageBox.Show(
                        "Edita la blacklist en el Notepad que se abrió.\n\n" +
                        "• Un autor por línea\n" +
                        "• Las líneas con # son comentarios\n" +
                        "• Guarda y cierra para aplicar cambios\n\n" +
                        "Los cambios se aplicarán al recargar la grilla.",
                        "Editar Blacklist",
                        MessageBoxButtons.OK,
                        MessageBoxIcon.Information);
                    
                    Task.Delay(1000).ContinueWith(_ => 
                    {
                        this.Invoke((Action)(() => LoadBlacklistedAuthors()));
                    });
                }
                catch (Exception ex)
                {
                    MessageBox.Show($"Error abriendo blacklist:\n{ex.Message}", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
            };
            buttonsRow.Controls.Add(btnEditBlacklist);
            
            var btnClearCaches = new Button { Text = "🗑️ Borrar cachés", Width = 160, Height = 40, BackColor = Color.FromArgb(120, 60, 60), ForeColor = Color.White, FlatStyle = FlatStyle.Flat, Font = new Font("Segoe UI", 10, FontStyle.Bold) };
            btnClearCaches.Click += (s, e) =>
            {
                try
                {
                    var paths = new[] { @"c:\p2p\author_cache.json", @"c:\p2p\author_cache.json.gz", @"c:\p2p\SlskDown\search_cache.json" };
                    int deleted = 0;
                    foreach (var path in paths)
                    {
                        try { if (System.IO.File.Exists(path)) { System.IO.File.Delete(path); deleted++; } }
                        catch (Exception ex) { Log($"No se pudo borrar {path}: {ex.Message}"); }
                    }
                    Log($"Cachés borradas ({deleted} archivos)");
                    MessageBox.Show($"Cachés borradas ({deleted} archivos)", "Borrar cachés", MessageBoxButtons.OK, MessageBoxIcon.Information);
                }
                catch (Exception ex) { MessageBox.Show($"Error al borrar cachés: {ex.Message}", "Borrar cachés", MessageBoxButtons.OK, MessageBoxIcon.Error); }
            };
            buttonsRow.Controls.Add(btnClearCaches);
            
            actionsPanel.Controls.Add(buttonsRow);
            flowLayout.Controls.Add(actionsPanel);
            
            // Nota informativa
            var lblAutoSave = new Label 
            { 
                Text = "ℹ️ Los cambios se guardan automáticamente", 
                AutoSize = true, 
                ForeColor = Color.FromArgb(100, 200, 100),
                Font = new Font("Segoe UI", 10, FontStyle.Italic),
                Margin = new Padding(0, 20, 0, 20)
            };
            flowLayout.Controls.Add(lblAutoSave);
            
            scrollPanel.Controls.Add(flowLayout);
            parent.Controls.Add(scrollPanel);
        }
        
        private Panel CreateConfigSection(string title, Color titleColor)
        {
            var panel = new FlowLayoutPanel
            {
                AutoSize = true,
                FlowDirection = FlowDirection.TopDown,
                WrapContents = false,
                Margin = new Padding(0, 0, 0, 20),
                Padding = new Padding(0, 0, 0, 10)
            };
            
            var lblSection = new Label 
            { 
                Text = title, 
                AutoSize = true, 
                ForeColor = titleColor, 
                Font = new Font("Segoe UI", 10, FontStyle.Bold),
                Margin = new Padding(0, 0, 0, 10)
            };
            panel.Controls.Add(lblSection);
            
            return panel;
        }
        
        private FlowLayoutPanel CreateConfigRow(string labelText)
        {
            var row = new FlowLayoutPanel 
            { 
                AutoSize = true, 
                FlowDirection = FlowDirection.LeftToRight,
                Margin = new Padding(0, 5, 0, 5)
            };
            
            var lbl = new Label 
            { 
                Text = labelText, 
                Width = 200, 
                ForeColor = Color.LightGray,
                TextAlign = ContentAlignment.MiddleLeft
            };
            row.Controls.Add(lbl);
            
            return row;
        }
