using System;
using System.Threading.Tasks;
using SlskDown.Core;

namespace SlskDown.Examples
{
    /// <summary>
    /// Ejemplo de cómo integrar eMule con SlskDown
    /// Este código muestra cómo configurar y usar búsquedas multi-red
    /// </summary>
    public class EmuleIntegrationExample
    {
        private NetworkOrchestrator _orchestrator;
        private EmuleClient _emuleClient;
        private SoulseekClient _soulseekClient;

        /// <summary>
        /// Inicializa el sistema multi-red con Soulseek y eMule
        /// </summary>
        public async Task InitializeAsync()
        {
            // 1. Crear NetworkOrchestrator
            _orchestrator = new NetworkOrchestrator();

            // 2. Configurar Soulseek (ya existente)
            _soulseekClient = new SoulseekClient();
            await _soulseekClient.ConnectAsync("tu_usuario", "tu_password");
            
            var soulseekSearchProvider = new SoulseekSearchProvider(_soulseekClient);
            _orchestrator.RegisterSearchProvider(soulseekSearchProvider);

            // 3. Configurar eMule (NUEVO)
            _emuleClient = new EmuleClient(
                host: "127.0.0.1",
                port: 4712,
                password: "tu_password_ec" // Configurada en eMule/aMule
            );

            try
            {
                // Conectar a eMule
                await _emuleClient.ConnectAsync();
                Console.WriteLine("✅ eMule conectado");

                // Registrar proveedores de eMule
                var emuleSearchProvider = new EmuleSearchProvider(_emuleClient);
                var emuleDownloadProvider = new EmuleDownloadProvider(_emuleClient);
                
                _orchestrator.RegisterSearchProvider(emuleSearchProvider);
                _orchestrator.RegisterDownloadProvider(emuleDownloadProvider);

                Console.WriteLine("✅ eMule registrado en NetworkOrchestrator");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"⚠️ eMule no disponible: {ex.Message}");
                Console.WriteLine("Continuando solo con Soulseek...");
            }
        }

        /// <summary>
        /// Realiza una búsqueda en múltiples redes simultáneamente
        /// </summary>
        public async Task SearchMultiNetworkAsync(string query)
        {
            Console.WriteLine($"\n🔍 Buscando '{query}' en todas las redes...\n");

            // Suscribirse a resultados parciales
            _orchestrator.PartialResultsReceived += (sender, e) =>
            {
                Console.WriteLine($"📦 {e.Results.Count} resultados de {e.NetworkName}");
                
                foreach (var result in e.Results)
                {
                    Console.WriteLine($"   - {result.FileName} ({FormatSize(result.SizeBytes)}) - Red: {result.NetworkSource}");
                }
            };

            // Iniciar búsqueda multi-red
            var searchRequest = new SearchRequest
            {
                SearchId = Guid.NewGuid().ToString(),
                Query = query,
                MaxResults = 50,
                Timeout = TimeSpan.FromSeconds(30)
            };

            var results = await _orchestrator.SearchAsync(searchRequest);

            Console.WriteLine($"\n✅ Búsqueda completada: {results.TotalResults} resultados totales");
            Console.WriteLine($"   - Soulseek: {results.Results.Count(r => r.NetworkSource == "Soulseek")}");
            Console.WriteLine($"   - eMule: {results.Results.Count(r => r.NetworkSource == "eMule")}");
        }

        /// <summary>
        /// Descarga un archivo desde la red más rápida disponible
        /// </summary>
        public async Task DownloadWithFailoverAsync(SearchResult result)
        {
            Console.WriteLine($"\n⬇️ Descargando: {result.FileName}");
            Console.WriteLine($"   Red original: {result.NetworkSource}");

            try
            {
                // NetworkOrchestrator maneja automáticamente el failover
                // Si la descarga falla en una red, intenta en otra
                var downloadManager = new MultiNetworkDownloadManager(_orchestrator);
                
                var download = await downloadManager.StartDownloadAsync(
                    result,
                    destinationPath: @"c:\downloads\",
                    enableFailover: true // Habilitar failover automático
                );

                Console.WriteLine($"✅ Descarga iniciada: {download.DownloadId}");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"❌ Error: {ex.Message}");
            }
        }

        /// <summary>
        /// Ejemplo completo de uso
        /// </summary>
        public static async Task Main(string[] args)
        {
            var example = new EmuleIntegrationExample();

            try
            {
                // 1. Inicializar sistema multi-red
                await example.InitializeAsync();

                // 2. Realizar búsqueda en todas las redes
                await example.SearchMultiNetworkAsync("programming books");

                // 3. Descargar resultado (ejemplo)
                // await example.DownloadWithFailoverAsync(selectedResult);
            }
            catch (Exception ex)
            {
                Console.WriteLine($"❌ Error fatal: {ex.Message}");
            }

            Console.WriteLine("\nPresiona cualquier tecla para salir...");
            Console.ReadKey();
        }

        private string FormatSize(long bytes)
        {
            string[] sizes = { "B", "KB", "MB", "GB" };
            double len = bytes;
            int order = 0;
            while (len >= 1024 && order < sizes.Length - 1)
            {
                order++;
                len = len / 1024;
            }
            return $"{len:0.##} {sizes[order]}";
        }
    }
}
